ProtectedRoute 实现

// ProtectedRoute.jsx
import { Navigate } from 'react-router-dom';
import useAuthStore from '../store/authStore';

function ProtectedRoute({ children }) {
  const { isAuthenticated } = useAuthStore();
  
  if (!isAuthenticated) {
    return ;
  }
  
  return children;
}

export default ProtectedRoute;
